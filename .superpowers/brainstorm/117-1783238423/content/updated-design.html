更新後的 Model

class WorkRecord(models.Model):
    user = ForeignKey(User, on_delete=CASCADE)        # 綁定使用者
    store = ForeignKey(Store, on_delete=CASCADE)       # 店家
    date = DateField()                                 # 日期
    start_time = TimeField(null=True, blank=True)       # 上班
    end_time = TimeField(null=True, blank=True)         # 下班
    hours = DecimalField(max_digits=5, decimal_places=1)      # 工時
    hourly_wage = PositiveIntegerField(default=196)            # 時薪 (預設196)
    rate_multiplier = DecimalField(max_digits=3, decimal_places=1, default=1.0)  # 倍率下拉
    overtime_hours = DecimalField(max_digits=4, decimal_places=1, default=0)
    overtime_rate = DecimalField(max_digits=3, decimal_places=2, default=1.33)
    notes = TextField(blank=True)
